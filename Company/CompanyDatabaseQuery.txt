  부서는 임시로 3가지 부서로 제한 한다 : 인사,개발,경영
  부서 코드 :
	- 경영 : 101
	- 인사 : 102
	- 개발 : 103

CREATE DATABASE company;

CREATE TABLE salary_man(
  s_id int PRIMARY KEY,
  s_name varchar(20) NOT NULL default "not thing",
  s_department char(20) NOT NULL default "not thing",
  s_rank char(10) NOT NULL default "not thing",
  s_mail varchar(320) NOT NULL default "not thing",
  s_phoneNumber char(10) NOT NULL default "not thing",  
  s_salary int NOT NULL default 0, 
  Join_Date datetime default CURRENT_TIMESTAMP NOT NULL
);
