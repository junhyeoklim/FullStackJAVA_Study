CREATE TABLE resign (
  id INT auto_increment PRIMARY key,
  s_date date ,
  s_department varchar(10) NOT NULL default "nothing",
  s_gender varchar(6) NOT NULL default "not",
  s_rank char(10) NOT NULL default "nothing",
  s_birth CHAR(8) NOT NULL DEFAULT 'nothing',
  INDEX idx_s_department (s_department),
  INDEX idx_s_gender (s_gender),
  INDEX idx_s_rank (s_rank),
  INDEX idx_s_birth (s_birth)
);
