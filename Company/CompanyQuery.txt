CREATE DATABASE company;

CREATE TABLE salary_man(
  s_id int PRIMARY KEY,
  s_name varchar(20) NOT NULL default "nothing",
  s_birth INT NOT NULL default 0,
  s_gender varchar(8) NOT NULL default "nothing",
  s_department char(20) NOT NULL default "nothing",
  s_rank char(10) NOT NULL default "nothing",
  s_mail varchar(320) NOT NULL default "nothing",
  s_phoneNumber varchar(12) NOT NULL default "nothing",  
  s_salary int NOT NULL default 0, 
  Join_Date datetime default CURRENT_TIMESTAMP,
  updateTime timestamp DEFAULT current_timestamp ON UPDATE CURRENT_TIMESTAMP
);
